### 서브쿼리
* UNNEST
* PUSH_SUBQ
* FILTER

```sql
SELECT
    A.계좌번호, A.계좌명, A.고객번호, C.고객명, C.고객상태코드, C.변경일시
FROM 계좌 A,
     고객변경이력 C
WHERE A.개설일자 = ${OPEN_DT}
  AND C.고객번호 = A.고객번호
  AND C.변경일시 = (SELECT /*+ NO_UNNEST */ MAX(변경일시)
                FROM 고객변경이력 H
                WHERE A.고객번호     = H.고객번호
                  AND H.고객상태코드 != '01')
```
